How to wrap an image classification dataset#

In this how-to, we will show you how to wrap the CIFAR-10 dataset as a maite.protocols.image_classification.Dataset. CIFAR-10 is an image classification dataset made by by Alex Krizhevsky, Vinod Nair, and Geoffrey Hinton. It’s available both through Hugging Face Hub and torchvision.datasets, but this how-to uses Hugging Face Hub.

1. Load the CIFAR-10 dataset from Hugging Face Hub#

Load the modules needed for this how-to.

3. Creating a MAITE wrapper for the source dataset#

We create a class that implements the maite.protocols.image_classification.Dataset protocol.

A MAITE image classification compliant Dataset needs to have the following two methods : - __getitem__(index: int) to return the number of data elements in the dataset - __len__() to get the number of data elements in the dataset via () method

and an attribute: - metadata containing an ‘id’ str and a map from indexes to label

InputType, TargetType, DatumMetadataType

__getitem__ is the most involved, it returns a tuple the image ArrayLike, the target label and the datum metadata. The image ArrayLike is typically an numpy array or torch tensor with the shape (C, H, W). Both the dtype and value range of the image ArrayLike is not specified. The target label is a one-hot encoding of the label. The datum metadata is a dictionary that includes at least a str or int id for the datum.

class Cifar10Dataset:
    def __init__(
        self,
        cifar10_dataset_split: datasets.Dataset | None = None,
        split: Literal["train"] | Literal["test"] = "train",
    ):
        """Create a MAITE CIFAR-10 dataset from a datasets.Dataset CIFAR-10 dataset.

        Parameters
        ----------
        cifar10_dataset_split : datasets.Dataset | None, optional
            The dataset to wrap, typically the train or test dataset, by default None
        split : Literal[&quot;train&quot;] | Literal[&quot;test&quot;], optional
            If the `cifar10_dataset_split` is not given, load this split, by default "train"
        """

        if cifar10_dataset_split is None:
            # Load (and possibly download) the dataset
            cifar10_dataset_dict = datasets.load_dataset("uoft-cs/cifar10")

            # We have already validated that cifar10 is a DatasetDict, but we leave this assertion
            # as a sanity check and to assist typechecking.
            assert isinstance(cifar10_dataset_dict, datasets.DatasetDict)

            # Use the given training or test split.
            self.dataset = cifar10_dataset_dict[split]
        else:
            # Save the CIFAR-10 dataset given by the user. This is helpful if you want to
            # sample the dataset using the Huggingface api prior to using it.
            self.dataset = cifar10_dataset_split

        # Create a dictionary from label number to label name from the label metadata
        # in the underlying dataset.
        index2label = {
            i: label for i, label in enumerate(self.dataset.features["label"].names)
        }

        self.metadata: DatasetMetadata = {
            "id": "CIFAR-10",
            "index2label": index2label,
        }

        # When the wrapper loads the dataset from a specific train/test split, record it in the metadata.
        if cifar10_dataset_split is None:
            self.metadata['split'] = split

        # Save the number of classes used in the dataset.
        num_classes = self.dataset.features["label"].num_classes
        self.targets = np.eye(num_classes)

    def __len__(self) -> int:
        """Return the number of items in the dataset

        Returns
        -------
        int
            The number of items in the dataset
        """
        return len(self.dataset)

    def __getitem__(
        self, index: int
    ) -> tuple[npt.NDArray, npt.NDArray, DatumMetadataType]:

        # Looking up an item in the dataset will return a diction with two keys
        # - "img": PIL.PngImagePlugin.PngImageFile image mode=RGB size=32x32>,
        # - "label": int
        item = self.dataset[index]
        img_pil = item["img"]
        label = item["label"]

        # Convert the PIL image to a Numpy array.
        img_hwc = np.array(img_pil)  # shape (H, W, C)

        # Use MAITE array index convention for representing images is (channel, height, width)
        img_chw = img_hwc.transpose(2, 0, 1)

        # One hot encoded tensor indicating the class label for this image.
        target = self.targets[label, :]

        # CIFAR-10 does not have any extra metadata, we record only the index of this datum.
        metadata: DatumMetadataType = {"id": index}

        return img_chw, target, metadata

def print_datum(dataset, index):
    img_arr_chw,target,datum_metadata = dataset[index]
    print(f"Datum {datum_metadata['id']}")
    print(f"  Input Image Array: {str(img_arr_chw)[:30]}...")
    print(f"    shape={img_arr_chw.shape}")
    print(f"    dtype={img_arr_chw.dtype}")
    display(PIL.Image.fromarray(img_arr_chw.transpose(1,2,0)))
    print(f"  Target: {target}")
    label_index=np.argmax(target)
    print(f"    target index: {np.argmax(target)}")
    print(f"    target label: {dataset.metadata['index2label'][label_index]}")
    print("  Metadata:")
    print(f"    {datum_metadata}")
